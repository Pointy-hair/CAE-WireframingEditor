import {
    mxUtils,
    mxShape,
    mxStencil
} from '../misc/mxExport.js';
/**
 * @module Shapes
 */

mxUtils.extend(CheckboxShape, mxShape);

/**
 * @classdesc The shape for the Checkbox-element
 * @constructor
 * @extends mxShape
 * @see Checkbox
 */
function CheckboxShape(){
    var str = '&lt;shape name="checkbox" w="100" h="30" aspect="1">&lt;foreground>&lt;fontsize size="25"/>&lt;text str="Checkbox" x="25" y="25"/>&lt;fillstroke/>&lt;strokewidth width="5"/>&lt;path>&lt;move x="8" y="45"/>&lt;line x="12" y="50"/>&lt;line x="20" y="30"/>&lt;/path>&lt;fillstroke/>&lt;/foreground>&lt;background>&lt;strokecolor color="black"/>&lt;fillcolor color="#e3e3e4"/>&lt;rect x="0" y="25" w="25" h="30"/>&lt;fillstroke/>&lt;/background>&lt;/shape>';
    var xml = mxUtils.parseXml(str);
    var stencil = new mxStencil(xml.documentElement);
    mxShape.call(this, stencil);
}
export default CheckboxShape;
